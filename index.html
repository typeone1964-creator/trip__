  <script>

    const questions = [

      { q: '旅の朝、あなたはどんな風に目覚めたい？', a: ['自然・癒やし','都会・刺激','優雅・リラックス','自由・気まま'] },

      { q: '知らない路地に迷い込んだら？', a: ['探検する','地図を見る','人に聞く','写真を撮る'] },

      { q: '旅先の食事で重視することは？', a: ['名店','ローカル','雰囲気','食べ歩き'] },

      { q: '本を一冊持っていくなら？', a: ['歴史小説','漫画','哲学書','ガイドブック'] },

      { q: '旅の靴を選ぶなら？', a: ['スニーカー','おしゃれ靴','スリッパ','トレッキング'] },

      { q: '夜の過ごし方は？', a: ['バー','星空','マッサージ','夜景散策'] },

      { q: 'カバンの余白は？', a: ['空っぽ','準備万端','ミニマル','お土産用'] },

      { q: '行きたい時代は？', a: ['太古','中世','未来','子供時代'] },

      { q: '帰宅後どうなっていたい？', a: ['筋肉痛','賢く','潤い','活力満タン'] },

      { q: 'あなたにとって旅とは？', a: ['冒険','逃避','発見','交流'] }

    ];


    let index = 0;

    let answers = [];


    function checkStart() {

      const days = document.getElementById('days').value;

      const budget = document.getElementById('budget').value.trim();

      document.getElementById('startBtn').disabled = !(days && budget);

    }


    function startQuestions() {

      document.getElementById('step-basic').classList.add('hidden');

      document.getElementById('step-question').classList.remove('hidden');

      showQuestion();

    }


    function showQuestion() {

      const q = questions[index];

      document.getElementById('question-text').innerText = `Q${index + 1}. ${q.q}`;

      const choicesDiv = document.getElementById('choices');

      choicesDiv.innerHTML = '';

      q.a.forEach((text, i) => {

        const div = document.createElement('div');

        div.className = 'choice';

        div.innerText = text;

        div.onclick = () => selectAnswer(i);

        choicesDiv.appendChild(div);

      });

    }


    function selectAnswer(i) {

      answers.push(i);

      index++;

      if (index < questions.length) {

        showQuestion();

      } else {

        showResult();

      }

    }


    function showResult() {

      document.getElementById('step-question').classList.add('hidden');

      document.getElementById('step-result').classList.remove('hidden');


      const days = document.getElementById('days').value;

      const budget = document.getElementById('budget').value;


      const type = answers.filter(a => a === 0).length >= 3

        ? '五感で感じるネイチャー・ビースト'

        : '知的好奇心旺盛なタイムトラベラー';


      const place = type === '五感で感じるネイチャー・ビースト'

        ? '屋久島・大自然トレッキング'

        : '金沢・歴史とアートの旅';


      document.getElementById('result-text').innerHTML = `

        <p><strong>${type}</strong> タイプです。</p>

        <p>日程：${days} ／ 予算：${budget}</p>

        <p>おすすめの旅先：<strong>${place}</strong></p>

      `;

    }

  </script>
